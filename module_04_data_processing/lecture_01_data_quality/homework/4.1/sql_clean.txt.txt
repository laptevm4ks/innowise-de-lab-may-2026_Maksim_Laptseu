DELETE FROM
    silver.silver_employees
WHERE
    employee_id IS NULL;


DELETE FROM
    silver.silver_employees
WHERE ctid NOT IN (
    SELECT
        MIN(ctid)
    FROM
	silver.silver_employees
    GROUP BY
	employee_id
);


DELETE FROM
    silver.silver_employees e
WHERE NOT EXISTS (
    SELECT
	1 
    FROM
	silver.silver_sales s 
    WHERE
	s.employee_id = e.employee_id
);


UPDATE
    silver.silver_sales s
SET 
    shop_id = e.shop_id,
    city_id = e.city_id
FROM
    silver.silver_employees e
WHERE
    s.employee_id = e.employee_id;
