WITH shop_stats AS (
    SELECT 
        ct.country_name,
        sh.shop_id AS shop_id,
        sh.address AS shop_address,
        COUNT(s.sales_id) AS total_sales_count,
        SUM(s.total_price) AS total_sales_amount
    FROM sales s
    JOIN employees e USING (employee_id)
    JOIN cities ci USING (city_id)
    JOIN shops sh USING (city_id)
    JOIN countries ct USING (country_id)
    GROUP BY 1, 2, 3
    HAVING COUNT(s.sales_id) >= 2)
SELECT 
    country_name,
    shop_id,
    shop_address,
    total_sales_count,
    total_sales_amount,
    SUM(total_sales_amount) OVER (PARTITION BY country_name) AS country_total_sales,
    total_sales_amount / SUM(total_sales_amount) OVER (PARTITION BY country_name) AS country_sales_share,
    RANK() OVER (PARTITION BY country_name ORDER BY total_sales_amount DESC) AS country_shop_rank,
    SUM(total_sales_amount) OVER (
        PARTITION BY country_name 
        ORDER BY total_sales_amount DESC
    ) AS country_running_total
FROM shop_stats
ORDER BY country_name ASC, country_shop_rank ASC;
